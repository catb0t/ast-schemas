rewrites as     om_dequote 1 {2}
rewrites as     1 {2}
rewrites as     <apply eval to entire new form starting from (d) 2 position>

-- dequote (operator)

-- operation?          truthy: x must be defined and not an operand

input           operation? x     where x is free
rewrites as     choose x {} defined? choose {} x operand? x
rewrites as     choose x {} defined? choose {} x {}
rewrites as     choose x {} defined? x                -- *1 from here
rewrites as     choose x {} {}
rewrites as     {}


-- operation?   contd from *1

1*              where x is bound
rewrites as     choose x {} defined? x
rewrites as     choose x {} x
rewrites as     x
